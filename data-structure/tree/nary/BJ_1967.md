> Question

* Outputs the diameter of the weighted tree

```txt
Input:
12
1 2 3
1 3 2
2 4 5
3 5 11
3 6 9
4 7 1
4 8 7
5 9 15
5 10 4
6 11 6
6 12 10

Output: 45
```

```py
import sys
from collections import deque
input = sys.stdin.readline
def bfs(G, x):
  dq = deque([x])
  dist = [-1 for _ in range(n)]
  dist[x] = 0
  while dq:
    x = dq.popleft()
    for w, nx in G[x]:
      if dist[nx] == -1:
        dist[nx] = dist[x] + w
        dq.append(nx)
  return max(dist), dist.index(max(dist))
n = int(input())
G = [[] for _ in range(n)]
for i in range(n-1):
  x, y, w = map(int, input().split())
  G[x-1].append([w, y-1])
  G[y-1].append([w, x-1])
print(bfs(G, bfs(G, 0)[1])[0])
```
