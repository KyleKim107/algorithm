```cpp
#include <bits/stdc++.h>
using namespace std;

int st[100010][18], dep[100010], a, b, lg=17;
vector<int> G[100010];

void dfs(int v, int d){
  if(dep[v] != 0)  return;
  dep[v] = d;
  for (int i = 1; 1<<i <= d; i++)
    st[v][i] = st[st[v][i-1]][i-1];
  for (auto x : G[v])
    if(st[x][0]==0)
      st[x][0] = v, dfs(x, d+1);
}
int lca(int a, int b){
  if (dep[a] < dep[b])  swap(a,b);
  for(int i = lg; i>=0; i--)
    if (dep[a] - (1<<i) >= dep[b])
      a=st[a][i];
  for(int i = lg; i>=0; i--)
    if(st[a][i]!=st[b][i])
      a=st[a][i], b=st[b][i];
  return (a==b)? a: st[a][0];
}
int main(){
  int n;
  for(scanf("%d", &n); i < n - 1; i++){
    cin >> a >> b);
    G[a].push_back(b);
    G[b].push_back(a);
  }
  st[1][0] = 1, dfs(1, 0);
  for(scanf("%d", &n); n--; ) {
    cin >> a >> b;
    cout << lca(a,b) << endl;
  }
  return 0;
}
```

```py
N = int(input())
G = [[]for i in range(N+1)]
for i in range(N-1):
  a, b = map(int,input().split())
  G[a].append(b)
  G[b].append(a)
D, P = [1] + [0 for i in range(N)], [0 for i in range(N+1)]
st = [1]
while st:
  i=st.pop()
  for j in G[i]:
    if not D[j]:
      D[j] = D[i] + 1
      P[j] = i
      st.append(j)
for _ in range(int(input())):
  a, b = map(int,input().split())
  while D[a] != D[b]:
    if D[a] > D[b]:
      a = P[a]
    else:
      b = P[b]
  while a != b:
    a, b = P[a], P[b]
  print(a)
```
