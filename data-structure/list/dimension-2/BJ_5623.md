> Question

* Given sum of all the two numbers of sequence A, time, output A

```txt
Input:
4
0 3 6 7
3 0 5 6
6 5 0 9
7 6 9 0

Output: 2 1 4 5
```

{% tab title='BJ_5623.cpp' %}

```cpp
#include <iostream>

using namespace std;

int arr[2][1000];
int main() {
  int n;
  cin >> n;

  for (int i = 0; i < 2; i++)
    for (int j = 0; j < n; j++)
      cin >> arr[i][j];

  if (n == 2) {
    if (arr[0][1] == 2) {
      cout << "1 1";
      return 0;
    }
  }
  int a1 = (arr[0][1] + arr[0][2] - arr[1][2]) / 2;
  cout << a1 << ' ';
  for (int i = 1; i < n; i++)
    cout <<(int)( arr[0][i] - a1) << ' ';
  return 0;
}
```
