> Question

* Delete until first blank and print concatnated string with God

```txt
Input:
5
baek joon
koo sa ga
ac ka
yu ka ri ko
ke sa ki yo

Output:
godjoon
godsaga
godka
godkariko
godsakiyo
```

{% tab title='BJ_13163.cpp' %}

```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
string s;
int main() {
  cin >> n;
  while(n--) {
    getline(cin, s);
    s.replace(s.begin(), s.begin() + s.find_first_of(' '), "god");
    for(char c : s) if (c != ' ') cout << c;
    cout << '\n';
  }
}
```

for _ in range(int(input())):
  l = input().split()
  l[0] = 'god'
  print(''.join(l))
