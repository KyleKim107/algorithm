{% tab title='BJ_1920.cpp' %}

```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
set<int> se;

int main(){
  scanf("%d", &n);
  for(int x; n--; ){
    scanf("%d", &x);
    se.insert(x);
  }
  scanf("%d", &n);
  for(int x; n--; ){
    scanf("%d", &x);
    printf("%d\n", se.find(x) != se.end());
  }
}
```
