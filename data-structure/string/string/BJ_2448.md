{% tab title='BJ_2448.cpp' %}

```cpp
#include<bits/stdc++.h>
using namespace std;

int main() {
  int a;
  cin >> a;
  vector<string> star = {"  *  ", " * * ","*****"};
  a /= 3;

  while (a != 1){
    int N = star.size();
    for (int i = 0; i < N; i++)
      star.push_back(star[i] + " " + star[i]);
    for (int i = 0; i < N; i++)
      star[i] = string(N, ' ') + star[i] + string(N, ' ');
    a /= 2;
  }

  for (string str : star)
    cout << str << "\n";

  return 0;
}
```

```py
N = int(input())
star = ["*","* *","*****"]
k = 3

while k < N:
  star += [s.ljust(k*2) + s for s in star]
  k *= 2

for s in star:
  print(s.center(2*N-1))
```
