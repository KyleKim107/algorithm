> Question

* There are N employees and M jobs where each person can do at most 2 jobs
* Print number of jobs that can be done

```txt
Input:
5 5
2 1 2
2 1 2
2 1 2
2 4 5
0

Output: 4
```

def dfs(u):
  visited[u] = 1
  for nx in G[u//2]:
    if W[nx]==-1 or (not visited[W[nx]] and dfs(W[nx])):
      P[u] = nx
      W[nx] = u
      return 1
  return 0
n,m = map(int,input().split())
G = [[] for _ in range(n)]
for i in range(n):
  a,*b = map(int, input().split())
  for j in b:
    G[i].append(j - 1)
P, W = [-1]*(2*n), [-1]*m
res = 0
for i in range(2*n):
  if P[i]==-1:
    visited=[0]*2*n
    if dfs(i): res+=1
print(res)
