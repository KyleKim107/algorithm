from sys import stdin
from heapq import *
input = stdin.readline

def dijkstra(s,d):
  v[s] = 0
  pq = [[0,s]]
  while pq:
    cost, x = heappop(pq)
    for i in range(n):
      if 0 < G[x][i] < v[i] - cost:
        v[i] = G[x][i] + cost
        p[i] = x
        heappush(pq, [G[x][i] + cost, i])
  return -1 if v[d] == 1e9 else v[d]

def r(s,d):
  pq = [[v[d], d]]
  while pq:
    cost, x = heappop(pq)
    for i in range(n):
      if G[i][x] > 0 and v[i] == cost - G[i][x]:
        G[i][x] = -1
        heappush(pq, [v[i], i])

while 1:
  n, m = map(int,input().split())
  if n == m == 0:
    break
  G, p = [[-1] * n for _ in range(n)], [-1] * n
  s, d = map(int,input().split())
  for _ in range(m):
    z, x, c = map(int, input().split())
    G[z][x] = c
  v = [1e9] * n
  t = dijkstra(s,d)
  if t == -1:
    print(-1)
  else:
    r(s, d)
    v = [1e9] * n
    print(dijkstra(s,d))
