```py
import math

def recur(old_G):
  new_G=[]
  for i in range(3 * len(old_G)):
    if i // len(old_G) == 1:
      new_G.append(old_G[i % len(old_G)] + " " * len(old_G) + old_G[i % len(old_G)])
    else:
      new_G.append(old_G[i % len(old_G)] * 3)
  return new_G

G = ["*"]
i = 1
N = int(input())

while i != N:
  G = recur(G)
  i *= 3
for i in G:
  print(i)
```
