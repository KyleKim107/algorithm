> Problem

* Find maximum block with tetromino

![BJ_14500](images/20210304_200335.png)

```txt
Input:
5 5
1 2 3 4 5
5 4 3 2 1
2 3 4 5 6
6 5 4 3 2
1 2 1 2 1

Output:
19
```

{% endtab %}
{% tab title='BJ_14500.py' %}

```py
n, m = map(int, input().split())
a = [list(map(int, input().split())) for _ in range(n)]
b = [
  [(0,1), (1,0), (1,1)],
  [(0,1), (0,2), (0,3)],
  [(1,0), (2,0), (3,0)],
  [(0,1), (0,2), (1,0)],
  [(0,1), (0,2), (-1,2)],
  [(1,0), (1,1), (1,2)],
  [(0,1), (0,2), (1,2)],
  [(1,0), (2,0), (2,1)],
  [(0,1), (1,1), (2,1)],
  [(0,1), (1,0), (2,0)],
  [(1,0), (2,0), (2,-1)],
  [(1,0), (1,1), (2,1)],
  [(0,1), (1,0), (-1,1)],
  [(0,1), (1,0), (1,-1)],
  [(0,1), (1,1), (1,2)],
  [(0,1), (0,2), (1,1)],
  [(1,0), (1,1), (1,-1)],
  [(1,0), (2,0), (1,-1)],
  [(1,0), (1,1), (2,0)]]

def tetromino(x, y):
  ret = 0
  for i in range(19):
    cur = a[x][y]
    for j in range(3):
      try:
        nx = x+b[i][j][0]
        ny = y+b[i][j][1]
        cur += a[nx][ny]
      except IndexError:
        continue
    ret = max(ret, cur)
  return ret

ans = 0
for i in range(n):
  for j in range(m):
    ans = max(ans, tetromino(i, j))

print(ans)
```
