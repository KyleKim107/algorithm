```cpp
// Time : O(N)
#include <bits/stdc++.h>
using namespace std;
int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
  int tt; cin >> tt;
  while (tt--) {
    int n; cin >> n;
    vector<vector<int>> V(n, vector<int>(2));
    for (int i = 0; i < n; i++)
      for (int j = 0; j < 2; j++)
        cin >> V[i][j];
    vector<vector<int>> G(n);
    for (int i = 0; i < n - 1; i++) {
      int u, v; cin >> u >> v;
      --u; --v;
      G[u].push_back(v);
      G[v].push_back(u);
    }
    vector<vector<long long>> dp(n, vector<long long>(2, -1));
    function<void(int, int)> dfs = [&](int v, int pr) {
      dp[v][0] = dp[v][1] = 0;
      for (int u : G[v]) {
        if (u == pr) continue;
        dfs(u, v);
        for (int x = 0; x < 2; x++)
          dp[v][x] += max(dp[u][0] + abs(V[v][x] - V[u][0]), dp[u][1] + abs(V[v][x] - V[u][1]));
      }
    };
    dfs(0, -1);
    cout << max(dp[0][0], dp[0][1]) << '\n';
  }
  return 0;
}
```
