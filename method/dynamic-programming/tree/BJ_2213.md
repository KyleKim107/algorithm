> Question

* Given Tree with weight in vertices
* Find maximum independent set that maximize sum weight

```txt
Input:
7
10 30 40 10 20 20 70
1 2
2 3
4 3
4 5
6 2
6 7

Output:
140
1 3 5 7
```

{% tab title='BJ_2213.py' %}

```py
import sys
input = sys.stdin.readline
sys.setrecursionlimit(1000000000)

N = int(input())
V = [0] + list(map(int,input().split()))
G = [[] for i in range(N + 1)]
for i in range(N - 1):
  a, b = map(int, input().split())
  G[a].append(b)
  G[b].append(a)
dp = [[0, 0] for i in range(N + 1)] # dp[v][0]: without v, dp[v][1]: with v

def dfs(u, p):
  dp[u][1] = V[u]
  for v in G[u]:
    if v == p: continue
    dfs(v, u)
    dp[u][0] += max(dp[v][0], dp[v][1])
    dp[u][1] += dp[v][0]

def track(u, p, last):
  if not last:
    if dp[u][0] < dp[u][1]:
      l.append(u)
      last = True
  else:
    last = False
  for v in G[u]:
    if v != p:
      track(v, u, last)

dfs(1, 0)
print(max(dp[1]))

l = []
track(1, 0, False)
print(*sorted(l))
```
