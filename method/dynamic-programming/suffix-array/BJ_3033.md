> Question

* Print longest string that appears twice

```txt
Input:
11
sabcabcfabc

Output: 3
```

{% endtab %}
{% tab title='BJ_3033.py' %}

```py
input()
s = input()
N = len(s)

tmp, lcp = [0] * N, [0] * N
sa = [i for i in range(N)]
r = [ord(i) for i in s]
f = lambda m: r[m] if m < N else -1

L = 1
while L <= N:
  sa.sort(key=lambda x: (f(x), f(x+L)))
  rnk = 0
  tmp[sa[0]] = rnk

  for i in range(1, N):
    if f(sa[i]) != f(sa[i-1]) or f(sa[i]+L) != f(sa[i-1]+L):
      rnk += 1
    tmp[sa[i]] = rnk

  r = tmp[:]
  L <<= 1

for i in range(N):
  r[sa[i]] = i

ans = 0

L = 0
for i in range(N):
  if r[i]:
    j = sa[r[i] - 1]
    while i + L < N and j+L<N and s[i + L] == s[j + L]:
      L += 1

    lcp[r[i]] = L
    ans = max(ans, L)
    L -= bool(L)

print(ans)
```
