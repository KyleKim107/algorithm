> Question

* Given square 2D array, find maximum 2d rectangle sum

```txt
Input:
4
0 -2 -7 0 9 2 -6 2
-4 1 -4 1 -1

8 0 -2

Output: 15  # 9 2 / -4 1 / -1 8
```

{% endtab %}
{% tab title='BJ_6374.cpp' %}

```cpp
#include <stdio.h>
int map[200][200], b[200], cnt, max = -200;
int main() {
  int n, i, j, k;
  scanf("%d", &n);
  for (i = 1; i <= n; i++)
    for (j = 1; j <= n; j++) scanf("%d", &map[i][j]);

  for (i = 1; i <= n; i++) {
    for (j = 1; j <= n; j++) b[j] = 0;
    for (j = i; j <= n; j++) {
      cnt = 0;
      for (k = 1; k <= n; k++) {
        b[k] += map[k][j];
        cnt += b[k];
        if (max < cnt) max = cnt;
        if (cnt < 0) cnt = 0;
      }
    }
  }
  printf("%d", max);
  return 0;
}
```
