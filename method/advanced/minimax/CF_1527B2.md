```py
for _ in range (int(input())):
  n = int(input())
  s = input()
  c = s.count('0')
  if n % 2 and s[n//2]=='0':
    if c > 2:
      print("ALICE")
    elif c==2:
      print("DRAW")
    else:
      print("BOB")
  elif s == s[::-1]:
    print("BOB")
  else:
    print("ALICE")
```
