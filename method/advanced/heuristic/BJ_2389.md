> Question

* Print center and radius of minimum enclosing circle

```txt
Input:
3
1 1
2 2
3 3

Output:
2 2 1.4142135624
```

{% endtab %}
{% tab title='BJ_2389.py' %}

```py
n = int(input())
cx, cy = 0, 0
points=[]

for i in range(n):
  x, y = map(float, input().split())
  points.append((x,y))
  cx += x
  cy += y

cx, cy = cx / n, cy / n
ratio = 0.1

for i in range(60100):
  dist, idx = 0, -1
  for j in range(len(points)):
    x,y = points[j]
    d=(cx-x)**2+(cy-y)**2
    if dist < d:
      dist, idx = d, j

  cx += (points[idx][0]-cx) * ratio
  cy += (points[idx][1]-cy) * ratio
  ratio *= 0.999

print(cx, cy, dist**0.5)
```
