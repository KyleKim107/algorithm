> Question

* print A ** B modular C

```txt
Input: 10 11 12
Output: 4   # 10 ** 11 % 12
```

{% endtab %}
{% tab title='BJ_1629.cpp' %}

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  long long A, B, C, ans = 1;
  cin >> A >> B >> C;
  while (B > 0) {
    if (B & 1) ans = ans * A % C;
    A = A * A % C;
    B >>= 1;
  }
  cout << ans;
}
```
