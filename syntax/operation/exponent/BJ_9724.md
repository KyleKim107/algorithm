{% tab title='BJ_9724.py' %}

```py
from math import ceil

n = int(input())

for i in range(n):
  a, b = map(int,input().split())
  a, b = ceil(a ** (1 / 3)), int(b ** (1 / 3))
  print(f"Case #{i + 1}: {b - a + 1}")
```
