```cpp
#include <iostream>
using namespace std;

char op;
int ans, x;
int main() {
  cin >> ans >> op;
  while (1) {
    cin >> x;
    if (op == '+')ans += x;
    else if (op == '-') ans -= x;
    else if (op == '*') ans *= x;
    else if (op == '/') ans /= x;
    cin >> op;
    if (op == '=') break;
  }
  cout << ans;
}
```
