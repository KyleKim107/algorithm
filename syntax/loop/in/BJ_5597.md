{% tab title='BJ_5597.cpp' %}

```cpp
#include <iostream>
using namespace std;

int main() {
  bool check[31] = {0, };
  int temp;
  for(int i=0; i<28; i++){
     cin >> temp;
     check[temp] = 1;}
  for(int j=1; j<31; j++){
    if(check[j] == 0) cout << j << endl;
  }
}
```

a = [int(input()) for _ in range(28)]
for i in range(1, 31):
  if i not in a:
    print(i)
