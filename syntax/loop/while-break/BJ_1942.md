> Question

* From first line, # clock integers included in the interval entered over three lines is printed, which is multiple of 3

```txt
Input:
00:59:58 01:01:24
22:47:03 01:03:24
00:00:09 00:03:37

Output:
29
2727
70
```

{% tab title='BJ_1942.cpp' %}

```cpp
#include <bits/stdc++.h>
int main() {
  int s, a, b, c, d, e, f;
  for(; ~scanf("%d:%d:%d %d:%d:%d", &a, &b, &c, &d, &e, &f); ){
    s = 0
    while (true) {
      if((a + b + c) % 3 == 0) s++;
      if(a == d && b == e && c == f) break;
      c++;
      if(c == 60) c = 0, b ++;
      if(b == 60) b = 0, a ++;
      if(a == 24) a = 0;
    }
    printf("%d\n", s);
  }
}
```

```py
for t in range(3) :
  l, r = input().split()
  h1, m1, s1 = int(l[0:2]), int(l[3:5]), int(l[6:8])
  h2, m2, s2 = int(r[0:2]), int(r[3:5]), int(r[6:8])
  ans = 0
  while True:
    ans += int(h1 + m1 + s1) % 3 == 0)
    if h1 == h2 and m1 == m2 and s1 == s2 :
      break
    s1 += 1
    if s1 == 60 :
      m1 += 1
      s1 = 0
    if m1 == 60 :
      h1 += 1
      m1 = 0
    if h1 == 24 :
      h1 = 0
  print(ans)
```
